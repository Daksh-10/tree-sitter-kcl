================================================================================
assert statements with error message
================================================================================

assert 1==2 , "Wrong"

--------------------------------------------------------------------------------

(module
      (assert_statement
        (comparison_operator
          (integer)
          (integer))
        (string
          (string_start)
          (string_content)
          (string_end))))

================================================================================
assert statements with 'if' condition
================================================================================

assert 1 == 2 if True, "error message"

--------------------------------------------------------------------------------

(module
  (if_statement
    (true)
    (assert_statement
      (comparison_operator
        (integer)
        (integer))
      (string
        (string_start)
        (string_content)
        (string_end)))
  ))
